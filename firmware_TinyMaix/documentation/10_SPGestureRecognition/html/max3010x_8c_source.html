210 void MAX3010X_softReset(void) {
211  bitMask(MAX3010X_MODECONFIG, MAX3010X_RESET_MASK, MAX3010X_RESET);
212 
213  // Poll for bit to clear, reset is then complete
214  // Timeout after 100ms
215  // unsigned long startTime = millis();
216  unsigned long Time = 0;
217  while (Time < 100)
218  {
219  uint8_t response = readRegister8(MAX3010X_MODECONFIG);
220  if ((response & MAX3010X_RESET) == 0) break; //We're done!
221  DelayMs(1); //Let's not over burden the I2C bus
222  Time++;
223  }
224 }

289

291 
292  switch (slotNumber) {
293  case (1):
294  bitMask(MAX3010X_MULTILEDCONFIG1, MAX3010X_SLOT1_MASK, device);

296  case (2):

308  }

362 // Die Temperature
363 // Returns temp in C
364 float MAX3010X_readTemperature() {
365 
366  //DIE_TEMP_RDY interrupt must be enabled
367  //See issue 19: https://github.com/sparkfun/SparkFun_MAX3010x_Sensor_Library/issues/19
368 
369  // Step 1: Config die temperature register to take 1 temperature sample
370  writeRegister8(MAX3010X_DIETEMPCONFIG, 0x01);
371 
372  // Poll for bit to clear, reading is then complete
373  // Timeout after 100ms
374  unsigned long startTime = 0;
375  while (startTime < 100)
376  {
377  //uint8_t response = readRegister8(_i2caddr, MAX3010X_DIETEMPCONFIG); //Original way
378  //if ((response & 0x01) == 0) break; //We're done!
379 
380  //Check to see if DIE_TEMP_RDY interrupt is set
381  uint8_t response = readRegister8(MAX3010X_INTSTAT2);
382  if ((response & MAX3010X_INT_DIE_TEMP_RDY_ENABLE) > 0) break; //We're done!
383  DelayMs(1); //Let's not over burden the I2C bus
384  startTime++;
385  }
386  //TODO How do we want to fail? With what type of error?
387  //? if(millis() - startTime >= 100) return(-999.0);
388 
389  // Step 2: Read die temperature register (integer)
390  int8_t tempInt = readRegister8(MAX3010X_DIETEMPINT);
391  uint8_t tempFrac = readRegister8(MAX3010X_DIETEMPFRAC); //Causes the clearing of the DIE_TEMP_RDY interrupt
392 
393  // Step 3: Calculate temperature (datasheet pg. 23)
394  return (float)tempInt + ((float)tempFrac * 0.0625);
395 }

428 //Setup the sensor
429 //The MAX3010X has many settings. By default we select:
430 // Sample Average = 4
431 // Mode = MultiLED
432 // ADC Range = 16384 (62.5pA per LSB)
433 // Sample rate = 50
434 //Use the default setup if you are just getting started with the MAX3010X sensor
435 void MAX3010X_setup(uint8_t powerLevel, uint8_t sampleAverage, uint8_t ledMode, int sampleRate, int pulseWidth, int adcRange) {
436  MAX3010X_softReset(); //Reset all configuration, threshold, and data registers to POR values
437 
438  //FIFO Configuration
439  //-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-
440  //The chip will average multiple samples of same type together if you wish
441  if (sampleAverage == 1) MAX3010X_setFIFOAverage(MAX3010X_SAMPLEAVG_1); //No averaging per FIFO record
442  else if (sampleAverage == 2) MAX3010X_setFIFOAverage(MAX3010X_SAMPLEAVG_2);
443  else if (sampleAverage == 4) MAX3010X_setFIFOAverage(MAX3010X_SAMPLEAVG_4);
444  else if (sampleAverage == 8) MAX3010X_setFIFOAverage(MAX3010X_SAMPLEAVG_8);
445  else if (sampleAverage == 16) MAX3010X_setFIFOAverage(MAX3010X_SAMPLEAVG_16);
446  else if (sampleAverage == 32) MAX3010X_setFIFOAverage(MAX3010X_SAMPLEAVG_32);
447  else MAX3010X_setFIFOAverage(MAX3010X_SAMPLEAVG_4);
448 
449  //setFIFOAlmostFull(2); //Set to 30 samples to trigger an 'Almost Full' interrupt
450  MAX3010X_enableFIFORollover(); //Allow FIFO to wrap/roll over
451  //-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-
452 
453  //Mode Configuration
454  //-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-
455  if (ledMode == 3) MAX3010X_setLEDMode(MAX3010X_MODE_MULTILED); //Watch all three LED channels
456  else if (ledMode == 2) MAX3010X_setLEDMode(MAX3010X_MODE_REDIRONLY); //Red and IR
457  else MAX3010X_setLEDMode(MAX3010X_MODE_REDONLY); //Red only
458  activeLEDs = ledMode; //Used to control how many uint8_ts to read from FIFO buffer
459  //-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-
460 
461  //Particle Sensing Configuration
462  //-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-
463  if(adcRange < 4096) MAX3010X_setADCRange(MAX3010X_ADCRANGE_2048); //7.81pA per LSB
464  else if(adcRange < 8192) MAX3010X_setADCRange(MAX3010X_ADCRANGE_4096); //15.63pA per LSB
465  else if(adcRange < 16384) MAX3010X_setADCRange(MAX3010X_ADCRANGE_8192); //31.25pA per LSB
466  else if(adcRange == 16384) MAX3010X_setADCRange(MAX3010X_ADCRANGE_16384); //62.5pA per LSB
467  else MAX3010X_setADCRange(MAX3010X_ADCRANGE_2048);
468 
469  if (sampleRate < 100) MAX3010X_setSampleRate(MAX3010X_SAMPLERATE_50); //Take 50 samples per second
470  else if (sampleRate < 200) MAX3010X_setSampleRate(MAX3010X_SAMPLERATE_100);
471  else if (sampleRate < 400) MAX3010X_setSampleRate(MAX3010X_SAMPLERATE_200);
472  else if (sampleRate < 800) MAX3010X_setSampleRate(MAX3010X_SAMPLERATE_400);
473  else if (sampleRate < 1000) MAX3010X_setSampleRate(MAX3010X_SAMPLERATE_800);
474  else if (sampleRate < 1600) MAX3010X_setSampleRate(MAX3010X_SAMPLERATE_1000);
475  else if (sampleRate < 3200) MAX3010X_setSampleRate(MAX3010X_SAMPLERATE_1600);
476  else if (sampleRate == 3200) MAX3010X_setSampleRate(MAX3010X_SAMPLERATE_3200);
477  else MAX3010X_setSampleRate(MAX3010X_SAMPLERATE_50);

587 //Polls the sensor for new data
588 //Call regularly
589 //If new data is available, it updates the head and tail in the main struct
590 //Returns number of new samples obtained
591 uint16_t MAX3010X_check(void)
592 {
593  //Read register FIDO_DATA in (3-uint8_t * number of active LED) chunks
594  //Until FIFO_RD_PTR = FIFO_WR_PTR
595 
596  uint8_t readPointer = MAX3010X_getReadPointer();
597  uint8_t writePointer = MAX3010X_getWritePointer();
598 
599  int numberOfSamples = 0;
600 
601  //Do we have new data?
602  if (readPointer != writePointer)
603  {
604  //Calculate the number of readings we need to get from sensor
605  numberOfSamples = writePointer - readPointer;
606  if (numberOfSamples < 0) numberOfSamples += 32; //Wrap condition
607 
608  //We now have the number of readings, now calc uint8_ts to read
609  //For this example we are just doing Red and IR (3 uint8_ts each)
610  int uint8_tsLeftToRead = numberOfSamples * activeLEDs * 3;
611 
612  //Get ready to read a burst of data from the FIFO register
613  uint8_t transmit_data_buffer;
614  transmit_data_buffer = MAX3010X_FIFODATA;
615  I2CWrite(MAX30105_ADDRESS, &transmit_data_buffer, 1, true );
616  // _i2cPort->beginTransmission(MAX3010X_ADDRESS);
617  // _i2cPort->write(MAX3010X_FIFODATA);
618  // _i2cPort->endTransmission();
619 
620  //We may need to read as many as 288 uint8_ts so we read in blocks no larger than I2C_BUFFER_LENGTH
621  //I2C_BUFFER_LENGTH changes based on the platform. 64 uint8_ts for SAMD21, 32 uint8_ts for Uno.
622  //Wire.requestFrom() is limited to BUFFER_LENGTH which is 32 on the Uno
623  while (uint8_tsLeftToRead > 0)
624  {
625  int toGet = uint8_tsLeftToRead;
626  if (toGet > I2C_BUFFER_LENGTH)
627  {
628  //If toGet is 32 this is bad because we read 6 uint8_ts (Red+IR * 3 = 6) at a time
629  //32 % 6 = 2 left over. We don't want to request 32 uint8_ts, we want to request 30.
630  //32 % 9 (Red+IR+GREEN) = 5 left over. We want to request 27.
631 
632  toGet = I2C_BUFFER_LENGTH - (I2C_BUFFER_LENGTH % (activeLEDs * 3)); //Trim toGet to be a multiple of the samples we need to read
633  }
634 
635  uint8_tsLeftToRead -= toGet;
636 
637  //Request toGet number of uint8_ts from sensor
638  //_i2cPort->requestFrom(MAX3010X_ADDRESS, toGet);
639  uint8_t data_to_read_buffer[32];
640  uint8_t data = 0;
641  //data_to_read_buffer = reg;
642  I2CRead(MAX30105_ADDRESS, &data, 0, true , data_to_read_buffer, toGet, true );
643 
644  while (toGet > 0)
645  {
646  sense.head++; //Advance the head of the storage struct
647  sense.head %= STORAGE_SIZE; //Wrap condition
648 
649  uint8_t temp[sizeof(uint32_t)]; //Array of 4 uint8_ts that we will convert into long
650  uint32_t tempLong;
651 
652  //Burst read three uint8_ts - RED
653  temp[3] = 0;
654  temp[2] = data_to_read_buffer[0];
655  temp[1] = data_to_read_buffer[1];
656  temp[0] = data_to_read_buffer[2];
657 
658  //Convert array to long
659  memcpy(&tempLong, temp, sizeof(tempLong));
660 
661  tempLong &= 0x3FFFF; //Zero out all but 18 bits
662 
663  sense.red[sense.head] = tempLong; //Store this reading into the sense array
664 
665  if (activeLEDs > 1)
666  {
667  //Burst read three more uint8_ts - IR
668  temp[3] = 0;
669  temp[2] = data_to_read_buffer[3];
670  temp[1] = data_to_read_buffer[4];
671  temp[0] = data_to_read_buffer[5];
672 
673  //Convert array to long
674  memcpy(&tempLong, temp, sizeof(tempLong));
675 
676  tempLong &= 0x3FFFF; //Zero out all but 18 bits
677 
678  sense.IR[sense.head] = tempLong;
679  }
680 
681  if (activeLEDs > 2)
682  {
683  //Burst read three more uint8_ts - Green
684  temp[3] = 0;
685  temp[2] = data_to_read_buffer[6];
686  temp[1] = data_to_read_buffer[7];
687  temp[0] = data_to_read_buffer[8];
688 
689  //Convert array to long
690  memcpy(&tempLong, temp, sizeof(tempLong));
691 
692  tempLong &= 0x3FFFF; //Zero out all but 18 bits
693 
694  sense.green[sense.head] = tempLong;
695  }
696 
697  toGet -= activeLEDs * 3;
698  }
699 
700  } //End while (uint8_tsLeftToRead > 0)
701 
702  } //End readPtr != writePtr
703 
704  return (numberOfSamples); //Let the world know how much new data we found
705 }
