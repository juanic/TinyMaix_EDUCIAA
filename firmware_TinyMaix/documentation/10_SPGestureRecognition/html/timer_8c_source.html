72 void TimerInit(timer_config *timer_ini)
73 {
74  switch(timer_ini->timer)
75  {
76  case TIMER_A:
77  SystemCoreClockUpdate();
78  SysTick_Config(SystemCoreClock/TIMER_A_1ms_TICK);
79  SysTick->CTRL = SysTick_CTRL_CLKSOURCE_Msk | SysTick_CTRL_ENABLE_Msk; /*Stop Timer*/
80  timer_a = timer_ini;
81  pIsrTimerA=timer_ini->pFunc;
82  break;
83 
84  case TIMER_B:
85  Chip_RIT_Init(LPC_RITIMER);
86  Chip_RIT_SetTimerInterval(LPC_RITIMER,TIMER_B_1ms_TICK);
87  timer_b = timer_ini;
88  pIsrTimerB=timer_ini->pFunc;
89  break;
90 
91  case TIMER_C:
92  SystemCoreClockUpdate();
93  SysTick_Config(SystemCoreClock/TIMER_C_1us_TICK);
94  SysTick->CTRL = SysTick_CTRL_CLKSOURCE_Msk | SysTick_CTRL_ENABLE_Msk; /*Stop Timer*/
95  timer_a = timer_ini;
96  pIsrTimerA=timer_ini->pFunc;
97  break;
98  }
99 }

101 void SysTick_Handler(void)
102 {
103  uint16_t period = timer_a->period;
104  tick_timer_a ++;
105  if ( tick_timer_a == period)
106  {
107  TimerReset(TIMER_A);
108  pIsrTimerA();
109  }
110 }

112 void RIT_IRQHandler(void)
113 {
114  uint16_t period = timer_b->period; /*calc period in cents*/
115  tick_timer_b ++;
116  if ( tick_timer_b == period)
117  {
118  TimerReset(TIMER_B);
119  pIsrTimerB(); /* Call app function */
120  }
121  Chip_RIT_ClearInt(LPC_RITIMER);
122 }
