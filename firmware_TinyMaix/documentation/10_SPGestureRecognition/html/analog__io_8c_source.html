77 void AnalogInputInit(analog_input_config *inputs)
78 {
79  static ADC_CLOCK_SETUP_T configADC;
80 
81  configADC.adcRate=1000;
82  configADC.burstMode=DISABLE;
83  configADC.bitsAccuracy=ADC_10BITS;
84 
85  Chip_ADC_Init(LPC_ADC0,&configADC);
86  Chip_ADC_SetSampleRate(LPC_ADC0, &configADC,ADC_MAX_SAMPLE_RATE);
87 
88  if(!(inputs->input ^ CH0))
89  {
90  Chip_ADC_EnableChannel(LPC_ADC0,ADC_CH0,ENABLE);
91  if(inputs->pAnalogInput != NULL)
92  {
93  pIsrAnalogInput=inputs->pAnalogInput;
94  Chip_ADC_Int_SetChannelCmd(LPC_ADC0,ADC_CH0,ENABLE);
95  NVIC_EnableIRQ(ADC0_IRQn);

97  }

142 void AnalogInputReadPolling(uint8_t channel, uint16_t *value)
143 {
145  Chip_ADC_SetStartMode(LPC_ADC0, ADC_START_NOW, ADC_TRIGGERMODE_RISING);
146  switch(channel)
147  {
148  case CH0:
149  while (Chip_ADC_ReadStatus(LPC_ADC0, ADC_CH0, ADC_DR_DONE_STAT) != SET);
151  Chip_ADC_ReadValue(LPC_ADC0,ADC_CH0, value);
152  break;
153  case CH1:
154  while (Chip_ADC_ReadStatus(LPC_ADC0, ADC_CH1, ADC_DR_DONE_STAT) != SET);
156  Chip_ADC_ReadValue(LPC_ADC0,ADC_CH1, value);
157  break;
158  case CH2:
159  while (Chip_ADC_ReadStatus(LPC_ADC0, ADC_CH2, ADC_DR_DONE_STAT) != SET);
161  Chip_ADC_ReadValue(LPC_ADC0,ADC_CH2, value);
162  break;
163  case CH3:
164  while (Chip_ADC_ReadStatus(LPC_ADC0, ADC_CH3, ADC_DR_DONE_STAT) != SET);
166  Chip_ADC_ReadValue(LPC_ADC0,ADC_CH3, value);
167  break;
168  }
169 }
