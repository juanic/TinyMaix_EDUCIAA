101 int8_t SwitchesRead(void)
102 {

105  mask |= SWITCH_1;
106  if (!GPIORead(GPIO_TEC_2))

115 void SwitchActivInt(uint8_t sw, void *ptr_int_func)
116 {
117  switch(sw)
118  {
119  case SWITCH_1:
120  ptr_tec_int_func[0] = ptr_int_func;
121  GPIOActivInt(GPIOGP0, GPIO_TEC_1, Switch1Int, LOW_EDGE);
122  break;
123  case SWITCH_2:
124  ptr_tec_int_func[1] = ptr_int_func;
125  GPIOActivInt(GPIOGP1, GPIO_TEC_2, Switch2Int, LOW_EDGE);
126  break;
127  case SWITCH_3:
128  ptr_tec_int_func[2] = ptr_int_func;
129  GPIOActivInt(GPIOGP2, GPIO_TEC_3, Switch3Int, LOW_EDGE);
130  break;
131  case SWITCH_4:
132  ptr_tec_int_func[3] = ptr_int_func;
133  GPIOActivInt(GPIOGP3, GPIO_TEC_4, Switch4Int, LOW_EDGE);
134  break;
135  }
136 }

138 void SwitchesActivGroupInt(uint8_t switchs, void *ptr_int_func)
139 {
140  uint8_t n_act_sw = 0;
141  gpio_t sw_int[N_SW];
142  ptr_tec_group_int_func = ptr_int_func;
143  if (switchs & SWITCH_1)
144  {
145  sw_int[n_act_sw] = GPIO_TEC_1;
146  n_act_sw++;
147  }
148  if (switchs & SWITCH_2)
149  {
150  sw_int[n_act_sw] = GPIO_TEC_2;
151  n_act_sw++;
152  }
153  if (switchs & SWITCH_3)
154  {
155  sw_int[n_act_sw] = GPIO_TEC_3;
156  n_act_sw++;
157  }
158  if (switchs & SWITCH_4)
159  {
160  sw_int[n_act_sw] = GPIO_TEC_4;
161  n_act_sw++;
162  }
163  GPIOActivGroupInt(GPIOGROUPGP0, sw_int, n_act_sw, ptr_int_func, LOW_EDGE);
164 }
