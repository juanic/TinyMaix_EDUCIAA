352  DelayUs(10);

362  WriteLCD(&lcd_sleep_out);
363  DelayMs(5);
364  WriteLCD(&lcd_on);
365  DelayMs(10);
366  /* Start screen on White */
367  ILI9341Fill(ILI9341_WHITE);
368  return SUCCESS;

422 void ILI9341DrawChar(uint16_t x, uint16_t y, char data, Font_t* font, uint16_t foreground, uint16_t background)
423 {
424  static uint16_t i, j, k;
425  static uint16_t char_row;
426  static uint16_t lcd_x, lcd_y;
427  static int32_t bytes_count;
428  static uint8_t pixel[MAX_VALUE_SIZE];
429 
430  /* Set coordinates */
431  lcd_x = x;
432  lcd_y = y;
433 
434  /* If at the end of a line of display, go to new line and set x to 0 position */
435  if ((lcd_x + font->FontWidth) > lcd_orientation.width)
436  {
437  lcd_y += font->FontHeight;
438  lcd_x = 0;
439  }
440 
441  SetCursorPosition(lcd_x, lcd_y, lcd_x + font->FontWidth - 1, lcd_y + font->FontHeight - 1);
442 
443  /* Number of bytes to write. We have to write 2 bytes/pixel */
444  bytes_count = font->FontHeight * font->FontWidth * 2;
445 
446  /* Start writing LCD memory */
447  lcd_cmd_t lcd_write = {MEM_WRITE, NULL, NULL};
448  WriteLCD(&lcd_write);
449 
450  /* Draw font data */
451  /* go through character rows */
452  k = 0;
453  for (i = 0; i < font->FontHeight; i++)
454  {
455  /* each 16bits data of a font character draws a full row of that character */
456  char_row = font->data[(data - ' ') * font->FontHeight + i];
457  /* go through character columns */
458  for (j = 0; j < font->FontWidth; j++)
459  {
460  /* If exceed buffer size, send buffer */
461  if ((2 * j + i * font->FontWidth * 2 - k * MAX_VALUE_SIZE + 1) > MAX_VALUE_SIZE)
462  {
463  lcd_cmd_t lcd_pixels = {NULL, MAX_VALUE_SIZE, pixel};
464  WriteLCD(&lcd_pixels);
465  bytes_count -= MAX_VALUE_SIZE;
466  k++;
467  }
468  /* The n=FontWidth first bits of the 16bits row data draws the corresponding part of a character */
469  if (char_row & (MSK_BIT16 >> j))
470  {
471  /* if bit = 1, draw put foreground color */
472  pixel[2 * j + i * font->FontWidth * 2 - k * MAX_VALUE_SIZE] = HighByte(foreground);
473  pixel[2 * j + i * font->FontWidth * 2 - k * MAX_VALUE_SIZE + 1] = LowByte(foreground);
474  }
475  else
476  {
477  pixel[2 * j + i * font->FontWidth * 2 - k * MAX_VALUE_SIZE] = HighByte(background);
478  pixel[2 * j + i * font->FontWidth * 2 - k * MAX_VALUE_SIZE + 1] = LowByte(background);
479  }
480  }
481  }
482  /* Send the rest of the buffer */
483  lcd_cmd_t lcd_pixels = {NULL, bytes_count, pixel};
484  WriteLCD(&lcd_pixels);
485 }

591  }

596  }
597  else
598  {
599  y_grow = DOWN;
600  }
601 
602  /* Vertical or horizontal line */
603  if (x_dist == 0 || y_dist == 0)
604  {
605  Fill(x0, y0, x1, y1, color);
606  }
607  /* Diagonal line */
608  else
609  {
610  if(x_dist > y_dist)
611  {
612  error = x_dist / 2;
613  }
614  else
615  {
616  error = y_dist / 2;
617  }
618 
619  while (1)
620  {
621  /* Draw start point */
622  ILI9341DrawPixel(x0, y0, color);
623  /* Loop ends when start point reaches end point */
624  if (x0 == x1 || y0 == y1)
625  {
626  break;
627  }
628  error_2 = error;
629  /* Determine if line must grow in x direction */
630  if (error_2 > -x_dist)
631  {
632  error -= y_dist;
633  x0 += x_grow; /* Move start point */
634  }
635  /* Determine if line must grow in y direction */
636  if (error_2 < y_dist)
637  {
638  error += x_dist;
639  y0 += y_grow; /* Move start point */
640  }
641  }
642  }
643 }
