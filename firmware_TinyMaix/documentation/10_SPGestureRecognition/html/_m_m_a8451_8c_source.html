57 bool MMA8451IsAlive( void ){
58 
59  /* Check connection */
60  uint8_t device_id=0;
61  uint8_t data_to_read_buffer;
62  data_to_read_buffer = MMA8451_REG_WHOAMI;
63  I2CRead(MMA8451_DEFAULT_ADDRESS, &data_to_read_buffer, 1, true, &device_id, 1, true );
64  if (device_id != 0x1A)
65  {
66  /* No MMA8451 detected ... return false */
67  return false;
68  }
69  return true;
70 }

80 bool MMA8451Config( mma8451_config_t config ){
81 
82  uint8_t transmit_data_buffer[2], data_to_read_buffer, rst = 0x40;
83 
84  transmit_data_buffer[0] = MMA8451_REG_CTRL_REG2;
85  transmit_data_buffer[1] = 0x40;
86  I2CWrite(MMA8451_DEFAULT_ADDRESS, &transmit_data_buffer, 2, true ); //reset
87 
88  while(rst & 0x40)
89  {
90  data_to_read_buffer = MMA8451_REG_CTRL_REG2;
91  I2CRead(MMA8451_DEFAULT_ADDRESS, &data_to_read_buffer, 1, true, &rst, 1, true );
92  }
93 
94 
95  transmit_data_buffer[0] = MMA8451_REG_XYZ_DATA_CFG;
96  transmit_data_buffer[1] = config.range;
97  I2CWrite(MMA8451_DEFAULT_ADDRESS, &transmit_data_buffer, 2, true ); //Rango de medicion
98 
99  transmit_data_buffer[0] = MMA8451_REG_CTRL_REG2;
100  transmit_data_buffer[1] = 0x02;
101  I2CWrite(MMA8451_DEFAULT_ADDRESS, &transmit_data_buffer, 2, true ); // High Resolution
102 
103  transmit_data_buffer[0] = MMA8451_REG_CTRL_REG1;
104  transmit_data_buffer[1] = 0x03;
105  I2CWrite(MMA8451_DEFAULT_ADDRESS, &transmit_data_buffer, 2, true ); // Activate at max rate, low noise mode
106 
107  return (MMA8451IsAlive());
108 }
