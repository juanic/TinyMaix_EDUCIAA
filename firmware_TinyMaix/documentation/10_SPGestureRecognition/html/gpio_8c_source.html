160 void GPIOInit(gpio_t pin, io_t io)
161 {
162  Chip_GPIO_Init(LPC_GPIO_PORT);
163  Chip_SCU_PinMuxSet(gpio[pin].hwPort, gpio[pin].hwPin, gpio[pin].mode); /* Asocia el pin fisico del micro a un gpio */
164  Chip_GPIO_SetPinDIR(LPC_GPIO_PORT, gpio[pin].gpioPort, gpio[pin].gpioPin, io); /* Indica si el gpio sera de salida o entrada */
165  Chip_GPIO_SetPinState(LPC_GPIO_PORT, gpio[pin].gpioPort, gpio[pin].gpioPin, FALSE); /* Inicializa el gpio en 0 */
166 }

193 void GPIOActivInt(gpiogp_t gp, gpio_t pin, void *ptr_int_func, bool edge)
194 {
195  ptr_GPIO_int_func[gp] = ptr_int_func;
196 
197  Chip_SCU_GPIOIntPinSel(gp, gpio[pin].gpioPort, gpio[pin].gpioPin); /* Configura el canal de la interrupcion*/
198 
199  Chip_PININT_ClearIntStatus(LPC_GPIO_PIN_INT, PININTCH(gp)); /* Limpia el estado de la interrupcion*/
200  Chip_PININT_SetPinModeEdge(LPC_GPIO_PIN_INT, PININTCH(gp)); /* Interrupcion por flanco*/
201  if(edge)
202  Chip_PININT_EnableIntHigh(LPC_GPIO_PIN_INT, PININTCH(gp)); /* Interrupcion cuando el flanco es ascendente*/
203  else
204  Chip_PININT_EnableIntLow(LPC_GPIO_PIN_INT, PININTCH(gp)); /* Interrupcion cuando el flanco es descendente*/
205 
206  NVIC_ClearPendingIRQ(32+gp); /* Limpia la interrupcion de PIN_INTX_IRQn definido en cmsis_43xx.h*/
207  NVIC_EnableIRQ(32+gp); /* Habilita la interrupcion de PIN_INTX_IRQn definido en cmsis_43xx.h*/
208 }

210 void GPIOActivGroupInt(gpiogroupgp_t group_gp, gpio_t *pins, uint8_t n_pins, void *ptr_int_func, bool edge)
211 {
212  uint8_t i;
213  ptr_GPIO_group_int_func[group_gp] = ptr_int_func;
214  Chip_GPIOGP_SelectOrMode(LPC_GPIOGROUP, group_gp); /* Cualquier GPIO dispara la interrupcion*/
215  Chip_GPIOGP_SelectEdgeMode(LPC_GPIOGROUP, group_gp); /* Interrupcion por flanco*/
216  Chip_GPIOGP_ClearIntStatus(LPC_GPIOGROUP, group_gp); /* Limpia el estado de la interrupcion*/
217  for(i=1 ; i<n_pins ; i++)
218  {
219  if(edge)
220  Chip_GPIOGP_SelectHighLevel(LPC_GPIOGROUP, group_gp, gpio[pins[i]].gpioPort, 1<<gpio[pins[i]].gpioPin);/* Interrupcion por nivel alto*/
221  else
222  Chip_GPIOGP_SelectLowLevel(LPC_GPIOGROUP, group_gp, gpio[pins[i]].gpioPort, 1<<gpio[pins[i]].gpioPin);/* Interrupcion por nivel bajo*/
223  Chip_GPIOGP_EnableGroupPins(LPC_GPIOGROUP, group_gp, gpio[pins[i]].gpioPort, 1<<gpio[pins[i]].gpioPin); /* Habilita el pin para la interrupcion de grupo*/
224  }
225  NVIC_EnableIRQ(40+group_gp); /* Habilita la interrupcion de GINTn_IRQn definido en cmsis_43xx.h*/
226 }
