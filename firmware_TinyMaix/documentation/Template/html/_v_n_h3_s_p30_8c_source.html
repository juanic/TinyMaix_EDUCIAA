47 uint8_t VNH3SP30_setSpeed(int speed) {
48  // Ensure we do not reverse the ina and inb setting in case speed==0 to guarantee the motor
49  // will free run to a stop (if you reverse ina and inb setting the controller will issue a full brake)
50  if (speed>0 || (speed==0 && forward)) {
51  GPIOOn(_inaPin);
52  GPIOOff(_inbPin);
53  forward = true;
54  } else if (speed<0) {
55  GPIOOff(_inaPin);
56  GPIOOn(_inbPin);
57  speed = - speed;
58  forward = false;
59  }
60  if (speed>400) speed = 400;
61  speed = (forward ? speed : -speed);
62  PWMSetDutyCycle(_pwmPin, speed);
63  PWMOn();
64  return VNH3SP30_status();
65 }

67 uint8_t VNH3SP30_brake(int brakePower) {
68  if (brakePower<0) brakePower = 0;
69  if (brakePower>100) brakePower = 100;
70  GPIOOff(_inaPin);
71  GPIOOff(_inbPin);
72  PWMSetDutyCycle(_pwmPin, brakePower); // map 400 to 255
73  PWMOff();
74  speed = 0;
75  return VNH3SP30_status();
76 }
