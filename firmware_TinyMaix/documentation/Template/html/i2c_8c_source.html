79  uint8_t Init_I2c( uint32_t clockRateHz ){
80 
81  /* Configuracion de las lineas de SDA y SCL de la placa*/
82  Chip_SCU_I2C0PinConfig( I2C0_STANDARD_FAST_MODE );
83  /* Inicializacion del periferico*/
84  Chip_I2C_Init( I2C0 );
85  /* Seleccion de velocidad del bus*/
86  Chip_I2C_SetClockRate( I2C0, clockRateHz );
87  /* Configuracion para que los eventos se resuelvan por polliong
88  // (la otra opcion es por interrupcion)*/
89  Chip_I2C_SetMasterEventHandler( I2C0, Chip_I2C_EventHandlerPolling );
90 
91  return true;
92  }

94 uint8_t I2CRead( uint8_t i2cSlaveAddress,
95  uint8_t* dataToReadBuffer,
96  uint16_t dataToReadBufferSize,
97  bool sendWriteStop,
98  uint8_t* receiveDataBuffer,
99  uint16_t receiveDataBufferSize,
100  bool sendReadStop ){
101 
102  I2CM_XFER_T i2cData;
103 
104  i2cData.slaveAddr = i2cSlaveAddress;
105  i2cData.options = 0;
106  i2cData.status = 0;
107  i2cData.txBuff = dataToReadBuffer;
108  i2cData.txSz = dataToReadBufferSize;
109  i2cData.rxBuff = receiveDataBuffer;
110  i2cData.rxSz = receiveDataBufferSize;
111 
112  if( Chip_I2CM_XferBlocking( LPC_I2C0, &i2cData ) == 0 ) {
113  return FALSE;
114  }
115 
116  return TRUE;
117  }
