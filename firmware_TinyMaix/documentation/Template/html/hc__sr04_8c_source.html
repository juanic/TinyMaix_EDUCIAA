75 bool HcSr04Init(gpio_t echo, gpio_t trigger)
76 {
77  echo_st = echo;
78  trigger_st = trigger;
79 
81  GPIOInit(echo, GPIO_INPUT);
82  GPIOInit(trigger, GPIO_OUTPUT);
83 
84  return true;
85 }

87 uint16_t HcSr04ReadDistanceInCentimeters(void)
88 {
89  uint16_t distance = 0;
90 
91  GPIOOn(trigger_st);
92  DelayUs(10);
93  GPIOOff(trigger_st);
94  while(!GPIORead(echo_st));
95  do
96  {
97  DelayUs(1);
98  distance++;
99  if(distance > 23200)
100  return 400;
101  }
102  while(GPIORead(echo_st));
103  return (distance/32);
104 }
