208 uint8_t* ftos(uint8_t W, uint8_t D, float data_n)
209 {
210  float data = abs(data_n);
211  long Wdata = data, tmp;
212  static uint8_t buf[10]= {0};
213  static uint8_t buf_neg[10]= {0};
214  //uint8_t buf[10]={0};
215  // get the first whole number and convert it
216  uint8_t j =0;
217 
218  float dec = data - (long)data;
219  buf[j] = Wdata / Pow(10, W - 1) + '0';
220  tmp = Wdata % Pow(10, W - 1);
221 
222  //now get the rest of the whole numbers
223  for (uint8_t i = 1; i < W; i++)
224  {
225  long factor = Pow(10, W - 1 - i);
226  buf[i+j] = (tmp / factor) + '0';
227  tmp %= factor;
228  }
229 
230  buf[W] = '.'; // add the decimal point
231 
232  // now do the decimal numbers
233  for (uint8_t i = 0; i < D; i++)
234  {
235  dec *= 10;
236  buf[W + i + 1 ] = ((long)dec % 10) + '0';
237  }
238  // don't forget the NULL terminator
239  buf[W + D + 1] = 0;
240  if(data_n<0){
241  buf_neg[0]='-';
242  for (uint8_t i = 0; i < D+W+1; i++)
243  {
244  buf_neg[i+1] = buf[i];
245  }
246  return &buf_neg;
247  }
248  else{ return &buf;}
249 }

256 void Process(){
257  //LedToggle(LED_RGB_B);

259  #ifdef MMA8451
260  MMA_getRawData(&x,&y,&z);
261  MMA_getAcceleration(&x_G,&y_G,&z_G);

265  x_G = ReadXValue();
266  y_G = ReadYValue();
267  z_G = ReadZValue();
268  float mag=0;
269  arm_sqrt_f32((x_G*x_G) + (y_G*y_G) + (z_G*z_G), &mag);
270  cyclesElapsedSample = cyclesCounterRead();
271  usElapsedSample = cyclesCounterToUs(cyclesElapsedSample);
272  #endif
273  #ifdef FILTER
274  cyclesCounterReset();
275  arm_biquad_cascade_df2T_f32(&IIR1_x,&x_G,&x_G_filt,1);
276  arm_biquad_cascade_df2T_f32(&IIR1_y,&y_G,&y_G_filt,1);
277  arm_biquad_cascade_df2T_f32(&IIR1_z,&z_G,&z_G_filt,1);
278  cyclesElapsed = cyclesCounterRead();
279  usElapsed = cyclesCounterToUs(cyclesElapsed);
280  #endif
281  switch(state){
282  /* Datalogging for training section */
283  case RECORD:
284  switch(record_state){
285  case WAITING_REC:
286  if(motionDetected(x_G,y_G,z_G)){
287  record_state = LOGGING_REC;
288  }
289  break;
290  case LOGGING_REC:
291  LedToggle(LED_RGB_G);
292  stdioPrintf(UART_USB.port, "%s,", ftos(1, 2, x_G));
293  stdioPrintf(UART_USB.port, "%s,", ftos(1, 2, y_G));
294  stdioPrintf(UART_USB.port, "%s,", ftos(1, 2, z_G));
295  frame++;
296  if(frame > (NUM_SAMPLES - 2)){
297  record_state = END_REC;
298  }
299  break;
300  case END_REC:
301  LedOff(LED_RGB_G);
302  stdioPrintf(UART_USB.port, "%s,", ftos(1, 2, x_G));
303  stdioPrintf(UART_USB.port, "%s,", ftos(1, 2, y_G));
304  stdioPrintf(UART_USB.port, "%s\n", ftos(1, 2, z_G));
305  frame = 0;
306  record_state = WAITING_REC;
307  break;
308  }
309  break;
310  case DATALOGGING:
311  if(motionDetected(x_G,y_G,z_G)){
312  LedToggle(LED_RGB_G);
313  if(!time){
314  #ifdef FILTER
315  stdioPrintf(UART_USB.port, "%s,", ftos(1, 2, x_G_filt));
316  stdioPrintf(UART_USB.port, "%s,", ftos(1, 2, y_G_filt));
317  stdioPrintf(UART_USB.port, "%s,", ftos(1, 2, z_G_filt));
318  #endif
319  stdioPrintf(UART_USB.port, "%s,", ftos(1, 2, x_G));
320  stdioPrintf(UART_USB.port, "%s,", ftos(1, 2, y_G));
321  stdioPrintf(UART_USB.port, "%s,", ftos(1, 2, z_G));
322  stdioPrintf(UART_USB.port, "%s\n", ftos(1, 2, mag));
323  }
324  else{
325  stdioPrintf(UART_USB.port, "%s,", ftos(1, 2, x_G));
326  stdioPrintf(UART_USB.port, "%s,", ftos(1, 2, y_G));
327  stdioPrintf(UART_USB.port, "%s,", ftos(1, 2, z_G));
328  stdioPrintf(UART_USB.port, "%d,", cyclesElapsedSample);
329  stdioPrintf(UART_USB.port, "%s\n", ftos(2,2,usElapsedSample));
330  }
331  }
332  else{
333  LedOff(LED_RGB_G);
334  }
335  break;
336  /* Prediction section */
337  case INFERRING:
338  #ifdef INFERRINGML
339  switch(record_state){
340  case WAITING_REC:
341  if(motionDetected(x_G_filt,y_G_filt,z_G_filt)){
342  record_state = LOGGING_REC;
343  LedOff(LED_3);
344  }
345  break;

347  features[frame * NUM_AXES + 0] = x_G_filt;
348  features[frame * NUM_AXES + 1] = y_G_filt;
349  features[frame * NUM_AXES + 2] = z_G_filt;
350  frame++;
351  if(frame > (NUM_SAMPLES - 1)){
352  record_state = END_REC;
353  }
354  break;
355  case END_REC:
356  stdioPrintf(UART_USB.port, "%s\n", predictLabel(features));
357  frame = 0;
358  LedOn(LED_3);

360  break;
361  }
362  #else
