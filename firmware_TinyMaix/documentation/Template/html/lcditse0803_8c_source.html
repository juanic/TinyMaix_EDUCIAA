77 bool LcdItsE0803Init(void)
78 {
79  /* Configuration of pins of data*/
80  GPIOInit(GPIO_LCD_1, GPIO_OUTPUT);

82  GPIOInit(GPIO_LCD_3, GPIO_OUTPUT);

90  actual_value=0;
91  LcdItsE0803Write(actual_value);
92  return true;
93 };

99 bool LcdItsE0803BCDtoPin(uint8_t value)
100 {
101  GPIOState(GPIO_LCD_1, (value & (1<<0))>>0);
102  GPIOState(GPIO_LCD_2, (value & (1<<1))>>1);
103  GPIOState(GPIO_LCD_3, (value & (1<<2))>>2);
104  GPIOState(GPIO_LCD_4, (value & (1<<3))>>3);
105  return true;
106 }

108  bool LcdItsE0803Write(uint16_t value)
109  {
110  uint8_t units, tens, hundreds;
111 
112  if(value<1000)
113  {
114  actual_value = value;
115 
116  hundreds = value/100;
117  tens = (value-(hundreds*100))/10;
118  units = (value-(hundreds*100)-(tens*10));
119 
120  /* Write hundreds */
121  LcdItsE0803BCDtoPin(hundreds);
122  GPIOOn(GPIO_1);
123  GPIOOff(GPIO_1);
124 
125  /* Write tens */
126  LcdItsE0803BCDtoPin(tens);
127  GPIOOn(GPIO_3);
128  GPIOOff(GPIO_3);
129 
130  /* Write units */
131  LcdItsE0803BCDtoPin(units);
132  GPIOOn(GPIO_5);
133  GPIOOff(GPIO_5);
134 
135 
136  return true; /* return 1 for values lower than 999 */
137  }
138  else
139  return false; /* return 0 for values higher than 999 */
140  }
