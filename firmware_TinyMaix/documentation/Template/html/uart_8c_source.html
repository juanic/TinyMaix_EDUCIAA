98 uint32_t UartInit(serial_config *port)
99 {
100  switch(port->port)
101  {
102  case SERIAL_PORT_PC:
103 
104  Chip_UART_Init(LPC_USART2);
105  Chip_UART_SetBaud(LPC_USART2, port->baud_rate);
106 
107  Chip_UART_SetupFIFOS(LPC_USART2, UART_FCR_FIFO_EN | UART_FCR_TRG_LEV0);
108 
109  Chip_UART_TXEnable(LPC_USART2);
110 
111  Chip_SCU_PinMux(UART_USB_TXD_MUX_GROUP, UART_USB_TXD_MUX_PIN, MD_PDN, FUNC6); /* P7_1: UART2_TXD */
112  Chip_SCU_PinMux(UART_USB_RXD_MUX_GROUP, UART_USB_RXD_MUX_PIN, MD_PLN|MD_EZI|MD_ZI, FUNC6); /* P7_2: UART2_RXD */
113 
114  serial_pc=port->pSerial;
115 
116  if(serial_pc != NO_INT)
117  {
118  Chip_UART_IntEnable(LPC_USART2, UART_IER_RBRINT);
119  NVIC_EnableIRQ(USART2_IRQn);
120  }
121 
122  break;
123 
124  case SERIAL_PORT_P2_CONNECTOR:
125 
126  Chip_UART_Init(LPC_USART3);
127  Chip_UART_SetBaud(LPC_USART3, port->baud_rate);
128 
129  Chip_UART_SetupFIFOS(LPC_USART3, UART_FCR_FIFO_EN | UART_FCR_TRG_LEV0);
130 
131  Chip_UART_TXEnable(LPC_USART3);
132 
133  Chip_SCU_PinMux(RS232_TXD_MUX_GROUP, RS232_TXD_MUX_PIN, MD_PDN, FUNC2); /* P2_3: UART3_TXD */
134  Chip_SCU_PinMux(RS232_TXD_MUX_GROUP, RS232_RXD_MUX_PIN, MD_PLN|MD_EZI|MD_ZI, FUNC2); /* P2_4: UART3_RXD */
135 
136  serial_p2_connector=port->pSerial;
137 
138  if(serial_p2_connector!= NO_INT)
139  {
140  Chip_UART_IntEnable(LPC_USART3, UART_IER_RBRINT);
141  NVIC_EnableIRQ(USART3_IRQn);
142  }
143 
144 
145  break;
146 
147  case SERIAL_PORT_RS485:
148  Chip_UART_Init(LPC_USART0);
149  Chip_UART_SetBaud(LPC_USART0, port->baud_rate);
150 
151  Chip_UART_SetupFIFOS(LPC_USART0, UART_FCR_FIFO_EN | UART_FCR_TRG_LEV0);
152 
153  Chip_UART_TXEnable(LPC_USART0);
154 
155  Chip_SCU_PinMux(RS485_TXD_MUX_GROUP, RS485_TXD_MUX_PIN, MD_PDN, FUNC7); /* P9_5: UART0_TXD */
156  Chip_SCU_PinMux(RS485_RXD_MUX_GROUP, RS485_RXD_MUX_PIN, MD_PLN|MD_EZI|MD_ZI, FUNC7); /* P9_6: UART0_RXD */
157 
158  Chip_UART_SetRS485Flags(LPC_USART0, UART_RS485CTRL_DCTRL_EN | UART_RS485CTRL_OINV_1);
159 
160  Chip_SCU_PinMux(DIR_RS485_MUX_GROUP, DIR_RS485_MUX_PIN, MD_PDN, FUNC2); /* P6_2: UART0_DIR */
161 
162  serial_rs485=port->pSerial;
163 
164  if(serial_rs485 != NO_INT)
165  {
166 
167  Chip_UART_IntEnable(LPC_USART0, UART_IER_RBRINT);
168  NVIC_EnableIRQ(USART0_IRQn);
169  }
170 
171 
172  break;
173  }
174 return TRUE;
175 }
