88 /* Wait for interrupt */
89 void Chip_EEPROM_WaitForIntStatus(LPC_EEPROM_T *pEEPROM, uint32_t mask)
90 {
91  uint32_t status;
92  while (1) {
93  status = Chip_EEPROM_GetIntStatus(pEEPROM);
94  if ((status & mask) == mask) {
95  break;
96  }
97  }
98  Chip_EEPROM_ClearIntStatus(pEEPROM, mask);
99 }
